try { const stored = JSON.parse(localStorage.getItem(GALLERY_FOLDERS_KEY)); return (stored && stored.length > 0) ? stored : defaults; } catch(e) { return defaults; } } /** * Ouvre le modal pour ajouter une image (Fichier) */ function openAddGalleryModal() { const folders = getGalleryFolders(); if (!fileInput.files || !fileInput.files[0]) { alert("Veuillez sélectionner une image."); return; } const file = fileInput.files[0]; const reader = new FileReader(); reader.onload = function(e) { const newItem = { id: 'G-' + Date.now(), title: values.gallery_title || 'Sans titre', folder: values.gallery_folder, url: e.target.result, // Base64 pour le local date: new Date().toLocaleDateString('fr-FR') }; saveToGallery(newItem); }; reader.readAsDataURL(file); } }); } /** * Crée un nouveau dossier dans la galerie */ function manageGalleryFolders() { const name = prompt("Nom du nouveau dossier pour la galerie :"); renderGalleryFilters(); // Rafraîchir les boutons de filtrage showToast("📁 Dossier '" + newFolderName + "' créé"); } else { alert("Ce dossier existe déjà."); } } } /** * Enregistrement avec fallback Firebase */ function saveToGallery(item) { if (typeof fbReady !== 'undefined' && fbReady && fbDB) { fbDB.ref('gallery').push(item).then(() => { showToast("✅ Image ajoutée à Firebase"); closeAppModal(); }); } else { let items = JSON.parse(localStorage.getItem(GALLERY_ITEMS_KEY) || '[]'); items.push(item); localStorage.setItem(GALLERY_ITEMS_KEY, JSON.stringify(items)); renderGallery(); closeAppModal(); showToast("✅ Image ajoutée (Local)"); } } /** * Affiche les filtres de dossiers dynamiquement */ function renderGalleryFilters() { const container = document.getElementById('gallery-folder-filters'); } });
